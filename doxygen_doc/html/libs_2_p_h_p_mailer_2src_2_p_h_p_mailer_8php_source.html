1465 if (is_callable($patternselect) && !is_string($patternselect)) {
1466 return call_user_func($patternselect, $address);
1467 }
1468 //Reject line breaks in addresses; it's valid RFC5322, but not RFC5321
1469 if (strpos($address, "\n") !== false || strpos($address, "\r") !== false) {
1470 return false;
1471 }

1473 case 'pcre': //Kept for BC
1474 case 'pcre8':
1475 /*
1476 * A more complex and more permissive version of the RFC5322 regex on which FILTER_VALIDATE_EMAIL
1477 * is based.
1478 * In addition to the addresses allowed by filter_var, also permits:
1479 * * dotless domains: `a@b`
1480 * * comments: `1234 @ local(blah) .machine .example`
1481 * * quoted elements: `'"test blah"@example.org'`
1482 * * numeric TLDs: `a@b.123`
1483 * * unbracketed IPv4 literals: `a@192.168.0.1`
1484 * * IPv6 literals: 'first.last@[IPv6:a1::]'
1485 * Not all of these will necessarily work for sending!
1486 *
1487 * @copyright 2009-2010 Michael Rushton
1488 * Feel free to use and redistribute this code. But please keep this copyright notice.
1489 */
1490 return (bool) preg_match(
1491 '/^(?!(?>(?1)"?(?>\\\[ -~]|[^"])"?(?1)){255,})(?!(?>(?1)"?(?>\\\[ -~]|[^"])"?(?1)){65,}@)' .
1492 '((?>(?>(?>((?>(?>(?>\x0D\x0A)?[\t ])+|(?>[\t ]*\x0D\x0A)?[\t ]+)?)(\‍((?>(?2)' .
1493 '(?>[\x01-\x08\x0B\x0C\x0E-\'*-\[\]-\x7F]|\\\[\x00-\x7F]|(?3)))*(?2)\‍)))+(?2))|(?2))?)' .

1495 '(?2)")(?>(?1)\.(?1)(?4))*(?1)@(?!(?1)[a-z0-9-]{64,})(?1)(?>([a-z0-9](?>[a-z0-9-]*[a-z0-9])?)' .
1496 '(?>(?1)\.(?!(?1)[a-z0-9-]{64,})(?1)(?5)){0,126}|\[(?:(?>IPv6:(?>([a-f0-9]{1,4})(?>:(?6)){7}' .
1497 '|(?!(?:.*[a-f0-9][:\]]){8,})((?6)(?>:(?6)){0,6})?::(?7)?))|(?>(?>IPv6:(?>(?6)(?>:(?6)){5}:' .

1501 );

1504 * This is the pattern used in the HTML5 spec for validation of 'email' type form input elements.
1505 *
1506 * @see https://html.spec.whatwg.org/#e-mail-state-(type=email)
1507 */
1508 return (bool) preg_match(
1509 '/^[a-zA-Z0-9.!#$%&\'*+\/=?^_`{|}~-]+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}' .
1510 '[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$/sD',

1518 * 'eai' is an acronym for Email Address Internationalization.

1521 *
1522 * @see https://html.spec.whatwg.org/#e-mail-state-(type=email)

1949 {

2725 {

3654 if ($this->UseSMTPUTF8 && !("comment" === $position)) {

3657
3658 $matchcount = 0;
3659 switch (strtolower($position)) {
3660 case 'phrase':
3661 if (!preg_match('/[\200-\377]/', $str)) {

3676 case 'text':

3684 } else {

3686 }

3694 $maxlen = static::MAX_LINE_LENGTH - $overhead;

5139 public function DKIM_HeaderC($signHeader)
5140 {
5141 //Normalize breaks to CRLF (regardless of the mailer)
5142 $signHeader = static::normalizeBreaks($signHeader, self::CRLF);
5143 //Unfold header lines
5144 //Note PCRE \s is too broad a definition of whitespace; RFC5322 defines it as `[ \t]`
5145 //@see https://www.rfc-editor.org/rfc/rfc5322#section-2.2
5146 //That means this may break if you do something daft like put vertical tabs in your headers.
5147 $signHeader = preg_replace('/\r\n[ \t]+/', ' ', $signHeader);
5148 //Break headers out into an array
5149 $lines = explode(self::CRLF, $signHeader);
5150 foreach ($lines as $key => $line) {
5151 //If the header is missing a :, skip it as it's invalid
5152 //This is likely to happen because the explode() above will also split
5153 //on the trailing LE, leaving an empty line
5154 if (strpos($line, ':') === false) {
5155 continue;
5156 }
5157 list($heading, $value) = explode(':', $line, 2);
5158 //Lower-case header name
5159 $heading = strtolower($heading);
5160 //Collapse white space within the value, also convert WSP to space
5161 $value = preg_replace('/[ \t]+/', ' ', $value);
5162 //RFC6376 is slightly unclear here - it says to delete space at the *end* of each value
5163 //But then says to delete space before and after the colon.
5164 //Net result is the same as trimming both ends of the value.
5165 //By elimination, the same applies to the field name
5166 $lines[$key] = trim($heading, " \t") . ':' . trim($value, " \t");
5167 }
5168
5169 return implode(self::CRLF, $lines);
5170 }
